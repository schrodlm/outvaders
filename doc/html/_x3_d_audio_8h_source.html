12 NOTES:
13 1. Definition of terms:
14 LFE: Low Frequency Effect -- always omnidirectional.
15 LPF: Low Pass Filter, divided into two classifications:
16 Direct -- Applied to the direct signal path,
17 used for obstruction/occlusion effect.
18 Reverb -- Applied to the reverb signal path,
19 used for occlusion effect only.
20
21 2. Volume level is expressed as a linear amplitude scaler:
22 1.0f represents no attenuation applied to the original signal,
23 0.5f denotes an attenuation of 6dB, and 0.0f results in silence.
24 Amplification (volume > 1.0f) is also allowed, and is not clamped.
25
26 3. X3DAudio uses a left-handed Cartesian coordinate system with values
27 on the x-axis increasing from left to right, on the y-axis from
28 bottom to top, and on the z-axis from near to far.
29 Azimuths are measured clockwise from a given reference direction.
30
31 Distance measurement is with respect to user-defined world units.
32 Applications may provide coordinates using any system of measure
33 as all non-normalized calculations are scale invariant, with such
34 operations natively occurring in the user-defined world unit space.
35 Metric constants are supplied only as a convenience.
36 Distance is calculated using the Euclidean norm formula.
37
38 4. Only real values are permissible with functions using 32-bit
39 float parameters -- NAN and infinite values are not accepted.
40 All computation occurs in 32-bit precision mode. */

99 // Distance curve point:
100 // Defines a DSP setting at a given normalized distance.
101 typedef struct X3DAUDIO_DISTANCE_CURVE_POINT
102 {
103 FLOAT32 Distance; // normalized distance, must be within [0.0f, 1.0f]
104 FLOAT32 DSPSetting; // DSP control setting
105 } X3DAUDIO_DISTANCE_CURVE_POINT, *LPX3DAUDIO_DISTANCE_CURVE_POINT;
106
107 // Distance curve:
108 // A piecewise curve made up of linear segments used to
109 // define DSP behaviour with respect to normalized distance.
110 //
111 // Note that curve point distances are normalized within [0.0f, 1.0f].
112 // X3DAUDIO_EMITTER.CurveDistanceScaler must be used to scale the
113 // normalized distances to user-defined world units.
114 // For distances beyond CurveDistanceScaler * 1.0f,
115 // pPoints[PointCount-1].DSPSetting is used as the DSP setting.
116 //
117 // All distance curve spans must be such that:
118 // pPoints[k-1].DSPSetting + ((pPoints[k].DSPSetting-pPoints[k-1].DSPSetting) / (pPoints[k].Distance-pPoints[k-1].Distance)) * (pPoints[k].Distance-pPoints[k-1].Distance) != NAN or infinite values
119 // For all points in the distance curve where 1 <= k < PointCount.
120 typedef struct X3DAUDIO_DISTANCE_CURVE
121 {
122 X3DAUDIO_DISTANCE_CURVE_POINT* pPoints; // distance curve point array, must have at least PointCount elements with no duplicates and be sorted in ascending order with respect to Distance
123 UINT32 PointCount; // number of distance curve points, must be >= 2 as all distance curves must have at least two endpoints, defining DSP settings at 0.0f and 1.0f normalized distance
124 } X3DAUDIO_DISTANCE_CURVE, *LPX3DAUDIO_DISTANCE_CURVE;

126 // Cone:
127 // Specifies directionality for a single-channel emitter by
128 // scaling DSP behaviour with respect to the emitter's front orientation.
129 // This is modeled using two sound cones: an inner cone and an outer cone.
130 // On/within the inner cone, DSP settings are scaled by the inner values.
131 // On/beyond the outer cone, DSP settings are scaled by the outer values.
132 // If on both the cones, DSP settings are scaled by the inner values only.
133 // Between the two cones, the scaler is linearly interpolated between the
134 // inner and outer values. Set both cone angles to 0 or X3DAUDIO_2PI for
135 // omnidirectionality using only the outer or inner values respectively.
136 typedef struct X3DAUDIO_CONE
137 {
138 FLOAT32 InnerAngle; // inner cone angle in radians, must be within [0.0f, X3DAUDIO_2PI]
139 FLOAT32 OuterAngle; // outer cone angle in radians, must be within [InnerAngle, X3DAUDIO_2PI]
140
141 FLOAT32 InnerVolume; // volume level scaler on/within inner cone, used only for matrix calculations, must be within [0.0f, 2.0f] when used
142 FLOAT32 OuterVolume; // volume level scaler on/beyond outer cone, used only for matrix calculations, must be within [0.0f, 2.0f] when used
143 FLOAT32 InnerLPF; // LPF (both direct and reverb paths) coefficient scaler on/within inner cone, used only for LPF (both direct and reverb paths) calculations, must be within [0.0f, 1.0f] when used
144 FLOAT32 OuterLPF; // LPF (both direct and reverb paths) coefficient scaler on/beyond outer cone, used only for LPF (both direct and reverb paths) calculations, must be within [0.0f, 1.0f] when used
145 FLOAT32 InnerReverb; // reverb send level scaler on/within inner cone, used only for reverb calculations, must be within [0.0f, 2.0f] when used
146 FLOAT32 OuterReverb; // reverb send level scaler on/beyond outer cone, used only for reverb calculations, must be within [0.0f, 2.0f] when used
147 } X3DAUDIO_CONE, *LPX3DAUDIO_CONE;

161 // Emitter:
162 // Defines a 3D audio source, divided into two classifications:
163 //
164 // Single-point -- For use with single-channel sounds.
165 // Positioned at the emitter base, i.e. the channel radius
166 // and azimuth are ignored if the number of channels == 1.
167 //
168 // May be omnidirectional or directional using a cone.
169 // The cone originates from the emitter base position,
170 // and is directed by the emitter's front orientation.
171 //
172 // Multi-point -- For use with multi-channel sounds.
173 // Each non-LFE channel is positioned using an
174 // azimuth along the channel radius with respect to the
175 // front orientation vector in the plane orthogonal to the
176 // top orientation vector. An azimuth of X3DAUDIO_2PI
177 // specifies a channel is a LFE. Such channels are
178 // positioned at the emitter base and are calculated
179 // with respect to pLFECurve only, never pVolumeCurve.
180 //
181 // Multi-point emitters are always omnidirectional,
182 // i.e. the cone is ignored if the number of channels > 1.
183 //
184 // Note that many properties are shared among all channel points,
185 // locking certain behaviour with respect to the emitter base position.
186 // For example, doppler shift is always calculated with respect to the
187 // emitter base position and so is constant for all its channel points.
188 // Distance curve calculations are also with respect to the emitter base
189 // position, with the curves being calculated independently of each other.
190 // For instance, volume and LFE calculations do not affect one another.
191 typedef struct X3DAUDIO_EMITTER
192 {
193 X3DAUDIO_CONE* pCone; // sound cone, used only with single-channel emitters for matrix, LPF (both direct and reverb paths), and reverb calculations, NULL specifies omnidirectionality
194 X3DAUDIO_VECTOR OrientFront; // orientation of front direction, used only for emitter angle calculations or with multi-channel emitters for matrix calculations or single-channel emitters with cones for matrix, LPF (both direct and reverb paths), and reverb calculations, must be normalized when used
195 X3DAUDIO_VECTOR OrientTop; // orientation of top direction, used only with multi-channel emitters for matrix calculations, must be orthonormal with OrientFront when used
196
197 X3DAUDIO_VECTOR Position; // position in user-defined world units, does not affect Velocity
198 X3DAUDIO_VECTOR Velocity; // velocity vector in user-defined world units/second, used only for doppler calculations, does not affect Position
199
200 UINT32 ChannelCount; // number of sound channels, must be > 0
201 FLOAT32 ChannelRadius; // channel radius, used only with multi-channel emitters for matrix calculations, must be >= 0.0f when used
202 FLOAT32* pChannelAzimuths; // channel azimuth array, used only with multi-channel emitters for matrix calculations, contains positions of each channel expressed in radians along the channel radius with respect to the front orientation vector in the plane orthogonal to the top orientation vector, or X3DAUDIO_2PI to specify a LFE channel, must have at least ChannelCount elements, all within [0.0f, X3DAUDIO_2PI] when used
203
204 X3DAUDIO_DISTANCE_CURVE* pVolumeCurve; // volume level distance curve, used only for matrix calculations, NULL specifies the default curve: [0.0f,1.0f], [1.0f,0.0f]
205 X3DAUDIO_DISTANCE_CURVE* pLFECurve; // LFE level distance curve, used only for matrix calculations, NULL specifies the default curve: [0.0f,1.0f], [1.0f,0.0f]
206 X3DAUDIO_DISTANCE_CURVE* pLPFDirectCurve; // LPF direct-path coefficient distance curve, used only for LPF direct-path calculations, NULL specifies the default curve: [0.0f,0.0f], [1.0f,0.25f]
207 X3DAUDIO_DISTANCE_CURVE* pLPFReverbCurve; // LPF reverb-path coefficient distance curve, used only for LPF reverb-path calculations, NULL specifies the default curve: [0.0f,0.25f], [1.0f,0.25f]
208 X3DAUDIO_DISTANCE_CURVE* pReverbCurve; // reverb send level distance curve, used only for reverb calculations, NULL specifies the default curve: [0.0f,1.0f], [1.0f,0.0f]
209
210 FLOAT32 CurveDistanceScaler; // curve distance scaler, used to scale normalized distance curves to user-defined world units and/or exaggerate their effect, does not affect any other calculations, must be within [FLT_MIN, FLT_MAX] when used
211 FLOAT32 DopplerScaler; // doppler shift scaler, used to exaggerate doppler shift effect, does not affect any other calculations, must be within [0.0f, FLT_MAX] when used
212 } X3DAUDIO_EMITTER, *LPX3DAUDIO_EMITTER;

215 // DSP settings:
216 // Receives results from a call to X3DAudioCalculate() to be sent
217 // to the low-level audio rendering API for 3D signal processing.
218 //
219 // The user is responsible for allocating the matrix coefficient table,
220 // delay time array, and initializing the channel counts when used.
221 typedef struct X3DAUDIO_DSP_SETTINGS
222 {
223 FLOAT32* pMatrixCoefficients; // [in] matrix coefficient table, receives an array representing the volume level of each source channel present in each destination channel with the source channels being the column index and the destination channels being the row index of the table, must have at least SrcChannelCount*DstChannelCount elements
224 FLOAT32* pDelayTimes; // [in] delay time array, receives delays for each destination channel in milliseconds, must have at least DstChannelCount elements (stereo final mix only)
225 UINT32 SrcChannelCount; // [in] number of source channels, must equal number of channels on respective emitter
226 UINT32 DstChannelCount; // [in] number of destination channels, must equal number of channels on the final mix
227
228 FLOAT32 LPFDirectCoefficient; // [out] LPF direct-path coefficient
229 FLOAT32 LPFReverbCoefficient; // [out] LPF reverb-path coefficient
230 FLOAT32 ReverbLevel; // [out] reverb send level
231 FLOAT32 DopplerFactor; // [out] doppler shift factor, scales resampler ratio for doppler shift effect, where the effective frequency = DopplerFactor * original frequency
232 FLOAT32 EmitterToListenerAngle; // [out] emitter-to-listener interior angle, expressed in radians with respect to the emitter's front orientation
233
234 FLOAT32 EmitterToListenerDistance; // [out] distance in user-defined world units from the emitter base to listener position, always calculated
235 FLOAT32 EmitterVelocityComponent; // [out] component of emitter velocity vector projected onto emitter->listener vector in user-defined world units/second, calculated only for doppler
236 FLOAT32 ListenerVelocityComponent; // [out] component of listener velocity vector projected onto emitter->listener vector in user-defined world units/second, calculated only for doppler
237 } X3DAUDIO_DSP_SETTINGS, *LPX3DAUDIO_DSP_SETTINGS;
