98 // DESCRIPTION:
99 // Calculates DSP settings with respect to 3D parameters:
100 //
101 // REMARKS:
102 // The number of bits set in SpeakerChannelMask should equal the number of
103 // channels expected on the final mix.
104 //
105 // PARAMETERS:
106 // SpeakerChannelMask - [in] speaker geometry configuration on the final mix, specifies assignment of channels to speaker positions, defined as per WAVEFORMATEXTENSIBLE.dwChannelMask, must be != 0
107 // Currently only (SPEAKER_FRONT_LEFT | SPEAKER_FRONT_RIGHT) is supported by XACT.
108 // pListener - [in] point of 3D audio reception
109 // pEmitter - [in] 3D audio source
110 // pDSPSettings - [out] receives calculation results, applied to an XACT cue via XACT3DApply
111 // pEngine - [in] pointer to the XACT engine
112 //
113 // RETURN VALUE:
114 // HResult error code
116 EXTERN_C HRESULT inline XACT3DCalculate (UINT32 SpeakerChannelMask, const X3DAUDIO_LISTENER* pListener, X3DAUDIO_EMITTER* pEmitter, X3DAUDIO_DSP_SETTINGS* pDSPSettings, IXACTEngine* pEngine)
117 {
118 XACTVARIABLEINDEX xactSpeedOfSoundID = pEngine->GetGlobalVariableIndex("SpeedOfSound");
119 XACTVARIABLEVALUE nSpeedOfSound;
120 HRESULT hr = pEngine->GetGlobalVariable(xactSpeedOfSoundID, &nSpeedOfSound);
121
122 if (SUCCEEDED(hr)) {
123 X3DAUDIO_HANDLE X3DInstance;
124 X3DAudioInitialize(SpeakerChannelMask, nSpeedOfSound, X3DInstance);
125
126 if (pEmitter->ChannelCount > 1 && pEmitter->pChannelAzimuths == NULL) {
127 pEmitter->ChannelRadius = 1.0f;
128
129 switch (pEmitter->ChannelCount) {
130 case 2: pEmitter->pChannelAzimuths = (float*)&aStereoLayout[0]; break;
131 case 3: pEmitter->pChannelAzimuths = (float*)&a2Point1Layout[0]; break;
132 case 4: pEmitter->pChannelAzimuths = (float*)&aQuadLayout[0]; break;
133 case 5: pEmitter->pChannelAzimuths = (float*)&a5Point1Layout[0]; break;
134 case 7: pEmitter->pChannelAzimuths = (float*)&a7Point1Layout[0]; break;
135 default: hr = E_FAIL; break;
136 }
137 }
138
139 X3DAUDIO_DISTANCE_CURVE_POINT DefaultCurvePoints[2] = { 0.0f, 1.0f, 1.0f, 1.0f };
140 X3DAUDIO_DISTANCE_CURVE DefaultCurve = { (X3DAUDIO_DISTANCE_CURVE_POINT*)&DefaultCurvePoints[0], 2 };
141 if (pEmitter->pVolumeCurve == NULL) {
142 pEmitter->pVolumeCurve = &DefaultCurve;
143 }
144 if (pEmitter->pLFECurve == NULL) {
145 pEmitter->pLFECurve = &DefaultCurve;
146 }
147
148 if (SUCCEEDED(hr)) {
149 X3DAudioCalculate(X3DInstance, pListener, pEmitter, X3DAUDIO_CALCULATE_MATRIX | X3DAUDIO_CALCULATE_DOPPLER | X3DAUDIO_CALCULATE_EMITTER_ANGLE, pDSPSettings);
150 }
151 }
152
153 return hr;
154 }

158 // DESCRIPTION:
159 // Applies a 3D calculation returned by XACT3DCalculate to a cue:
160 //
161 // PARAMETERS:
162 // pDSPSettings - [in] calculation results generated by XACT3DCalculate
163 // pCue - [in] cue to which to apply pDSPSettings
164 // ChannelCount - [in] should be set equal to pDSPSettings->SrcChannelCount.
165 //
166 // RETURN VALUE:
167 // HResult error code
169 EXTERN_C HRESULT inline XACT3DApply (X3DAUDIO_DSP_SETTINGS* pDSPSettings, IXACTCue* pCue, UINT32 ChannelCount)
170 {
171 HRESULT hr = pCue->SetMatrixCoefficients(ChannelCount, 2, pDSPSettings->pMatrixCoefficients);
172
173 if SUCCEEDED(hr) {
174 XACTVARIABLEINDEX xactDistanceID = pCue->GetVariableIndex("Distance");
175 hr = pCue->SetVariable(xactDistanceID, pDSPSettings->EmitterToListenerDistance);
176 }
177 if SUCCEEDED(hr) {
178 XACTVARIABLEINDEX xactDopplerID = pCue->GetVariableIndex("DopplerPitchScalar");
179 hr = pCue->SetVariable(xactDopplerID, pDSPSettings->DopplerFactor);
180 }
181 if SUCCEEDED(hr) {
182 XACTVARIABLEINDEX xactOrientationID = pCue->GetVariableIndex("OrientationAngle");
183 hr = pCue->SetVariable(xactOrientationID, pDSPSettings->EmitterToListenerAngle * (180.0f / X3DAUDIO_PI));
184 }
185
186 return hr;
187 }
